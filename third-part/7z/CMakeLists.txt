		set(ENABLE_7Z_ASM 0)
	elseif(${CMAKE_GENERATOR_PLATFORM} MATCHES ARM)
		set(ENABLE_7Z_ASM 0)
	elseif(MSVC_VERSION LESS 1600 AND CMAKE_CL_64)

	else()
		set(ENABLE_7Z_ASM 1)
